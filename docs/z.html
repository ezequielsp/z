    Add an attribute to an element.

    	z.attr = function(el, obj)
	{
		for(i in obj) {
			var attribute = obj[i][0];
			var val       = obj[i][1];
			if(!val) { 
				el.getAttribute(attribute)
			} else {
				el.setAttribute(attribute, val);
			} ;

		}

		return el;
	};
